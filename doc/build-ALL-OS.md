# Build instructions for all OS

## Ubuntu / Debian

sudo apt-get install -y libboost-system-dev libboost-filesystem-dev libboost-chrono-dev libboost-test-dev libboost-thread-dev

sudo apt-get install software-properties-common

sudo add-apt-repository ppa:luke-jr/db48

sudo apt-get update

sudo apt-get install -y build-essential libtool autotools-dev automake pkg-config libssl-dev libevent-dev bsdmainutils python3

sudo apt-get install -y libdb4.8-dev libdb4.8++-dev
